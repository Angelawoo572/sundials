  cvadj_mem = CVadjMalloc (cvode_mem, 150, CV_HERMITE);

Why doesn't CVodeF provide intermediate output in the same way as
CVode?  Is this a bug in CVodeF or am I misunderstanding how things
are supposed to work?  If it is my error, then what is the appropriate
way to get intermediate output when using CVodeF?

Thanks,

jwe



#include <stdio.h>

#include "cvodea.h"
#include "nvector_serial.h"
#include "cvodes_dense.h"

static int
f (realtype t, N_Vector x, N_Vector xdot, void *f_data)
{
  NV_Ith_S(xdot,0) = -NV_Ith_S(x,0) + 1.0;

  return(0);
}

int
main (int argc, char *argv[])
{
  void *cvadj_mem;

  realtype reltol = 1e-6, abstol = 1e-6;

  realtype time = 0.0;

  int flag, ncheck;

  int i;

  void *cvode_mem = CVodeCreate (CV_BDF, CV_NEWTON);

  N_Vector x = N_VNew_Serial (1);
  NV_Ith_S(x,0) = 0.0;

  flag = CVodeMalloc (cvode_mem, f, 0.0, x, CV_SS, reltol, &abstol);

  flag = CVDense (cvode_mem, 1);

  cvadj_mem = CVadjMalloc (cvode_mem, 150, CV_HERMITE);

  for (i = 0; i < 11; i++)
    {
      realtype tout = (i+1)*0.1;
#ifdef USE_CVODEF_AND_ADJOINT_MEM
      flag = CVodeF (cvadj_mem, tout, x, &time, CV_NORMAL, &ncheck);
#else
      flag = CVode (cvode_mem, tout, x, &time, CV_NORMAL);
#endif
      fprintf (stderr, "i = %2d, tout = %2.1f, flag = %d, x = %g\n",
               i, tout, flag, NV_Ith_S(x,0));
    }

  CVodeFree (&cvode_mem);
  N_VDestroy_Serial (x); 
  CVadjFree (&cvadj_mem);

  return 0;
}
