#include <stdio.h>

#include "cvodea.h"
#include "nvector_serial.h"
#include "cvodes_dense.h"

static int
f (realtype t, N_Vector x, N_Vector xdot, void *f_data)
{
  NV_Ith_S(xdot,0) = -NV_Ith_S(x,0) + 1.0;

  return 0;
}

static int
fb (realtype t, N_Vector x, N_Vector xb, N_Vector xbdot, void *f_data)
{
  NV_Ith_S(xbdot,0) = NV_Ith_S(xb,0) + 1.0;

  return 0;
}

int
main (int argc, char *argv[])
{
  void *cvadj_mem;

  realtype reltol = 1e-6, abstol = 1e-6;

  realtype time = 0.0;

  int flag, ncheck;

  int i;

  void *cvode_mem = CVodeCreate (CV_BDF, CV_NEWTON);

  N_Vector x = N_VNew_Serial (1);

  N_Vector xb = N_VNew_Serial (1);

  NV_Ith_S(x,0) = 0.0;

  NV_Ith_S(xb,0) = 0.0;

  CVodeMalloc (cvode_mem, f, 0.0, x, CV_SS, reltol, &abstol);

  CVDense (cvode_mem, 1);

  cvadj_mem = CVadjMalloc (cvode_mem, 150, CV_HERMITE);

  fprintf (stderr, "forward integration:\n");

  flag = CVodeF (cvadj_mem, 5.0, x, &time, CV_NORMAL, &ncheck);

  fprintf (stderr, "tout = %2.1f, flag = %d, x = %g\n",
           time, flag, NV_Ith_S(x,0));

  CVodeCreateB (cvadj_mem, CV_BDF, CV_NEWTON);

  CVodeMallocB (cvadj_mem, fb, 5.0, xb, CV_SS, reltol, &abstol);

  CVDenseB (cvadj_mem, 1);

  fprintf (stderr, "backard integration:\n");

#ifdef INTERMEDIATE_OUTPUT
  for (i = 0; i < 50; i++)
    {
      realtype tout = 5.0 - (1+i)*0.1;

      flag = CVodeB (cvadj_mem, tout, xb, &time, CV_NORMAL);

      fprintf (stderr, "tout = %2.1f, flag = %d, xb = %g\n",
               time, flag, NV_Ith_S(xb,0));
    }
#else
  flag = CVodeB (cvadj_mem, 0.0, xb, &time, CV_NORMAL);

  fprintf (stderr, "tout = %2.1f, flag = %d, xb = %g\n",
           time, flag, NV_Ith_S(xb,0));
#endif

  CVodeFree (&cvode_mem);
  N_VDestroy_Serial (x); 
  N_VDestroy_Serial (xb); 
  CVadjFree (&cvadj_mem);

  return 0;
}
