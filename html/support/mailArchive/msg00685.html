static int runCVode(void *cvode_mem, N_Vector y, realtype *t, realtype tout, Userdata data, FILE *fl)
{
  int flag, ql;
  realtype hl, tn;
  flag=CVodeGetCurrentTime(cvode_mem, &tn);
  if(check_flag(&flag, "CVodeGetCurrentTime", 1)) return(1);
  flag=CVodeGetLastStep(cvode_mem, &hl);
  if(check_flag(&flag, "CVodeGetLastStep", 1)) return(1);
  if((tout-*t)*hl<0.0){
    flag=CVodeReInit(cvode_mem, f, *t, y, CV_SS, data->reltol, &data->abstol);
    if(check_flag(&flag, "CVodeReInit", 1)) return(1);}
  if(*t<tout)
    while(*t<tout){
      flag=CVode(cvode_mem, tout, y, t, CV_ONE_STEP);
      if(check_flag(&flag, "CVode", 1)) return(1);
      flag=CVodeGetCurrentTime(cvode_mem, &tn);
      if(check_flag(&flag, "CVodeGetCurrentTime", 1)) return(1);
      flag=CVodeGetLastStep(cvode_mem, &hl);
      if(check_flag(&flag, "CVodeGetLastStep", 1)) return(1);
      flag=CVodeGetLastOrder(cvode_mem, &ql);
      if(check_flag(&flag, "CVodeGetLastOrder", 1)) return(1);
      fprintf(fl, "Time %le, Step %le Order %d\n", tn, hl, ql);}
    else
      while(*t>tout){
      flag=CVode(cvode_mem, tout, y, t, CV_ONE_STEP);
      if(check_flag(&flag, "CVode", 1)) return(1);
      flag=CVodeGetCurrentTime(cvode_mem, &tn);
      if(check_flag(&flag, "CVodeGetCurrentTime", 1)) return(1);
      flag=CVodeGetLastStep(cvode_mem, &hl);
      if(check_flag(&flag, "CVodeGetLastStep", 1)) return(1);
      flag=CVodeGetLastOrder(cvode_mem, &ql);
      if(check_flag(&flag, "CVodeGetLastOrder", 1)) return(1);
      fprintf(fl, "Time %le, Step %le Order %d\n", tn, hl, ql);}
  return(flag);
}

int f(realtype t, N_Vector y, N_Vector ydot, void *f_data)
{
  Userdata data;
  data=(Userdata)f_data;
  int i, j, k;
  realtype r, r2;
  realtype *y_data, *ydot_data;
  y_data=NV_DATA_S(y);
  ydot_data=NV_DATA_S(ydot);
  for(i=0;i<NUM_ASTER;i++){
    for(j=0;j<3;j++)
      ydot_data[6*i+j]=y_data[6*i+3+j];
    for(j=0;j<3;j++)
      ydot_data[6*i+3+j]=0.0;
    for(j=0;j<NUM_ASTER;j++){
      if(j==i) continue;
      r2=0.0;
      for(k=0;k<3;k++)
        r2+=(y_data[6*i+k]-y_data[6*j+k])*(y_data[6*i+k]-y_data[6*j+k]);
      r=sqrt(r2);
      for(k=0;k<3;k++)
        ydot_data[6*i+3+k]+=data->p[7*j+6]*(y_data[6*j+k]-y_data[6*i+k])/(r2*r);}}
  return(0);
}

/*
 * Check function return value...
 *   opt == 0 means SUNDIALS function allocates memory so check if
 *            returned NULL pointer
 *   opt == 1 means SUNDIALS function returns a flag so check if
 *            flag >= 0
 *   opt == 2 means function allocates memory so check if returned
 *            NULL pointer 
 */

static int check_flag(void *flagvalue, char *funcname, int opt)
{
  int *errflag;
  /* Check if SUNDIALS function returned NULL pointer - no memory allocated */
  if(opt==0&&flagvalue==NULL){
    fprintf(stderr, "\nSUNDIALS_ERROR: %s() failed - returned NULL pointer\n\n",
	    funcname);
    return(1);}
  /* Check if flag < 0 */
  else if(opt==1){
    errflag=(int *)flagvalue;
    if(*errflag<0){
      fprintf(stderr, "\nSUNDIALS_ERROR: %s() failed with flag = %d\n\n",
	      funcname, *errflag);
      return(1);}}
  /* Check if function returned NULL pointer - no memory allocated */
  else if(opt==2&&flagvalue==NULL){
    fprintf(stderr, "\nMEMORY_ERROR: %s() failed - returned NULL pointer\n\n",
	    funcname);
    return(1);}
  return(0);
}
