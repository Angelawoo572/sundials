[sundials-users] Step sizes for Jacobian approximation

From Jens Bastian <Jens.Bastian@eas.iis.fraunhofer.de>
Date Wed, 31 May 2006 15:46:03 +0200 
Hello,

when using IDA I detected a problem when the initial values are 0 and
atol is small. IDACalcIC() stops with

[IDA ERROR]  IDACalcIC
  The residual routine or the linear setup or solve routine had a
recoverable error, but IDACalcIC was unable to recover.

An example showing this behavior is attached.

I figured out that the reason for this is the calculation of the step
sizes for the approximation of the Jacobian by difference quotients. For
initial values 0 it holds |y(j)|=0 and |yp(j)|=0 and therefore the
stepsize is sigma(j)=sqrt(epsmach)*abs(j). Obviously sigma(j) becomes
too small for small values of the absolute error.

I changed the line
    inc = srur*MAX(ABS(yj),MAX( ABS(hh*ypj), ONE/ewt_data[j]));
in ida_dense.c into
    inc = MAX(srur*(ABS(yj), ABS(hh*ypj)), ONE/ewt_data[j]);
and now the example works.

Best regards
Jens

/*
 */

#include <stdio.h>
#include <math.h>

#include "ida.h"
#include "nvector_serial.h"
#include "ida_dense.h"
#include "sundials_types.h"
#include "sundials_math.h"

/* Problem Constants */

#define NEQ   2

/* Macro to define dense matrix elements, indexed from 1. */

#define IJth(A,i,j) DENSE_ELEM(A,i-1,j-1)

/* Prototypes of functions called by IDA */

int res(realtype tres, N_Vector yy, N_Vector yp, N_Vector resval, void *rdata);
int jac(long int Neq, realtype tt, N_Vector yy, N_Vector yp,
        N_Vector resvec, realtype cj, void *jdata, DenseMat JJ,
        N_Vector tempv1, N_Vector tempv2, N_Vector tempv3);

/*
 *--------------------------------------------------------------------
 * Main Program
 *--------------------------------------------------------------------
 */

int main(void)
{
  void *mem;
  N_Vector yy, yp, id;
  realtype atol, rtol, *yval, *ypval, *idval;
  realtype t0, tout1, tout, tret;
  int iout, retval, retvalr;

  mem = NULL;
  yy = yp = NULL;
  yval = ypval = NULL;

  /* Allocate N-vectors. */
  yy = N_VNew_Serial(NEQ);
  yp = N_VNew_Serial(NEQ);
  id = N_VNew_Serial(NEQ);

  /* Create and initialize  y, y', and absolute tolerance vectors. */
  yval  = NV_DATA_S(yy);
  yval[0] = 0;
  yval[1] = 0;

  ypval = NV_DATA_S(yp);
  ypval[0] = 0;
  ypval[1] = 0;

  atol = 1e-10;
  rtol = 1e-6;

  idval = NV_DATA_S(id);
  idval[0] = 1;
  idval[1] = 0;

  /* Integration limits */
  t0 = 0;
  tout1 = 0.1;

  /* Call IDACreate and IDAMalloc to initialize IDA memory */
  mem = IDACreate();
  retval = IDAMalloc(mem, res, t0, yy, yp, IDA_SS, rtol, &atol);

  retval = IDASetId(mem, id);

  /* Call IDADense and set up the linear solver. */
  retval = IDADense(mem, NEQ);

  retval = IDACalcIC(mem, t0, yy, yp, IDA_YA_YDP_INIT, tout1);

  printf("%f\t%25.20g\t%15.20g\n", t0, yval[0], yval[1]);

  return 0; 
}

/*
 *--------------------------------------------------------------------
 * Functions called by IDA
 *--------------------------------------------------------------------
 */

/*
 * Define the system residual function. 
 */

int res(realtype tres, N_Vector yy, N_Vector yp, N_Vector rr, void *rdata)
{
  realtype *yval, *ypval, *rval;

  yval = NV_DATA_S(yy); 
  ypval = NV_DATA_S(yp); 
  rval = NV_DATA_S(rr);

  rval[0] = ypval[0] - yval[1];
  rval[1] = yval[1] - exp(tres);

  printf("t = %f\ny[0] = %25.20g\ty[1] = %25.20g\ny'[0] = %25.20g\ty'[1] = 
%25.20g\nr[0] = %25.20g\tr[1] = %25.20g\n\n", tres, yval[0], yval[1], ypval[0], 
ypval[1], rval[0], rval[1]);

  return(0);
}
